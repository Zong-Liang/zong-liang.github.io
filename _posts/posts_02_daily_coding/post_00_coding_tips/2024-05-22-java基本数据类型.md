Java 中有八种基本数据类型。

## Java 代码

```java
/**
 * @author zongliang_i
 */
public class Main {

    public static void main(String[] args) {

        // --- 1. 整型 (Integer Types) ---
        // 用于表示没有小数部分的数值。

        // byte: 8位，占用1个字节，范围从 -128 到 127
        byte byteVar = 100;
        System.out.println("--- 整型 ---");
        System.out.println("byte 变量值: " + byteVar);
        System.out.println("byte 类型最小值: " + Byte.MIN_VALUE); // -128
        System.out.println("byte 类型最大值: " + Byte.MAX_VALUE); // 127
        System.out.println();

        // short: 16位，占用2个字节，范围从 -32,768 到 32,767
        short shortVar = 30000;
        System.out.println("short 变量值: " + shortVar);
        System.out.println("short 类型最小值: " + Short.MIN_VALUE); // -32768
        System.out.println("short 类型最大值: " + Short.MAX_VALUE); // 32767
        System.out.println();

        // int: 32位，占用4个字节，是Java中最常用的整数类型。
        // 范围从 -2,147,483,648 到 2,147,483,647
        int intVar = 2000000000;
        System.out.println("int 变量值: " + intVar);
        System.out.println("int 类型最小值: " + Integer.MIN_VALUE);
        System.out.println("int 类型最大值: " + Integer.MAX_VALUE);
        System.out.println();

        // long: 64位，占用8个字节，用于表示非常大的整数。
        // 声明long类型时，建议在数值末尾加上 'L' 或 'l'。
        long longVar = 9000000000000000000L;
        System.out.println("long 变量值: " + longVar);
        System.out.println("long 类型最小值: " + Long.MIN_VALUE);
        System.out.println("long 类型最大值: " + Long.MAX_VALUE);
        System.out.println();

        // 示例：整数溢出 (Integer Overflow)
        int maxInt = Integer.MAX_VALUE;
        System.out.println("int 最大值: " + maxInt);
        System.out.println("int 最大值 + 1 (溢出后): " + (maxInt + 1)); // 结果会变成int的最小值
        System.out.println("----------------------------------------\n");


        // --- 2. 浮点型 (Floating-Point Types) ---
        // 用于表示有小数部分的数值。

        // float: 32位，占用4个字节，单精度浮点数。
        // 声明float类型时，必须在数值末尾加上 'f' 或 'F'。
        float floatVar = 3.14159f;
        System.out.println("--- 浮点型 ---");
        System.out.println("float 变量值: " + floatVar);
        System.out.println("float 类型最小值: " + Float.MIN_VALUE); // 最小正数
        System.out.println("float 类型最大值: " + Float.MAX_VALUE);
        System.out.println();

        // double: 64位，占用8个字节，双精度浮点数，是默认的浮点类型。
        double doubleVar = 3.141592653589793;
        System.out.println("double 变量值: " + doubleVar);
        System.out.println("double 类型最小值: " + Double.MIN_VALUE); // 最小正数
        System.out.println("double 类型最大值: " + Double.MAX_VALUE);
        System.out.println();

        // 示例：浮点数精度问题 (Floating-Point Precision Issue)
        double d1 = 0.1;
        double d2 = 0.2;
        System.out.println("0.1 + 0.2 的实际结果: " + (d1 + d2)); // 结果不是精确的0.3
        System.out.println("----------------------------------------\n");


        // --- 3. 字符型 (Character Type) ---
        // char: 16位，占用2个字节，用于表示单个Unicode字符。
        // 字符需要用单引号 ' ' 括起来。
        char charVar1 = 'A';
        char charVar2 = '中';
        // 也可以直接使用Unicode值
        char charVar3 = '\u0041'; // 'A' 的Unicode编码

        System.out.println("--- 字符型 ---");
        System.out.println("char 变量1 (字母): " + charVar1);
        System.out.println("char 变量2 (汉字): " + charVar2);
        System.out.println("char 变量3 (Unicode): " + charVar3);
        System.out.println("char 类型最小值 (Unicode码): " + (int) Character.MIN_VALUE); // 0
        System.out.println("char 类型最大值 (Unicode码): " + (int) Character.MAX_VALUE); // 65535

        // 示例：char 与 int 的关系
        int asciiValue = (int) charVar1; // 将char转换为int以获取其ASCII/Unicode值
        System.out.println("字符 'A' 对应的整数值: " + asciiValue); // 输出 65
        System.out.println("----------------------------------------\n");


        // --- 4. 布尔型 (Boolean Type) ---
        // boolean: 只有两个值：true 和 false。
        boolean booleanVarTrue = true;
        boolean booleanVarFalse = false;

        System.out.println("--- 布尔型 ---");
        System.out.println("boolean 变量 (true): " + booleanVarTrue);
        System.out.println("boolean 变量 (false): " + booleanVarFalse);

        // 示例：在控制流程中使用boolean
        if (booleanVarTrue) {
            System.out.println("这是一个为 true 的条件判断。"); //
        }
        System.out.println("----------------------------------------\n");


        // --- 5. 数据类型转换 (Type Casting) ---
        System.out.println("--- 数据类型转换 ---");

        // 自动类型转换 (Widening Casting)
        // 从小范围类型到大范围类型的转换是自动的。
        // byte -> short -> int -> long -> float -> double
        int myInt = 100;
        double myDouble = myInt; // 自动将int转换为double
        System.out.println("自动类型转换 (int to double):");
        System.out.println("原始 int 值: " + myInt);
        System.out.println("转换后 double 值: " + myDouble);
        System.out.println();

        // 强制类型转换 (Narrowing Casting)
        // 从大范围类型到小范围类型的转换需要手动强制执行，并可能导致数据丢失。
        // double -> float -> long -> int -> short -> char -> byte
        double anotherDouble = 9.78;
        int anotherInt = (int) anotherDouble; // 强制将double转换为int
        System.out.println("强制类型转换 (double to int):");
        System.out.println("原始 double 值: " + anotherDouble);
        System.out.println("转换后 int 值 (小数部分被截断): " + anotherInt); //

        byte anotherByte = (byte) 130; // 130超出了byte的范围(-128到127)
        System.out.println("强制类型转换 (int to byte) - 范围溢出:");
        System.out.println("原始 int 值: 130");
        System.out.println("转换后 byte 值 (数据丢失): " + anotherByte); // 结果会因溢出而变得不正确
        System.out.println("----------------------------------------\n");
    }
}
```
